\begin{homeworkProblem}[Asymptotic Complexity Comparisons]
    Order the following functions so that for all $i,j$, if $f_i$ comes before 
    $f_j$ in the order then $f_i = O(f_j)$. Do not justify your answers.
    
    \begin{itemize}
        \item $f_1(n) = 3^{n}$
        \item $f_2(n) = n^{\frac{1}{3}}$
        \item $f_3(n) = 12$
        \item $f_4(n) = 2^{\log_2 n}$
        \item $f_5(n) = \sqrt{n}$
        \item $f_6(n) = 2^n$
        \item $f_7(n) = \log_2 n$
        \item $f_8(n) = 2^{\sqrt n}$
        \item $f_9(n) = n^3$
        \item $f_{10}(n) = \log_3 n$
        \item $f_{11}(n) = \log^2 n$
    \end{itemize}

    As an answer you may just write the functions as a list, e.g. $f_8, f_9, f_1, \ldots$
    \\

\end{homeworkProblem}

\begin{homeworkAnswer}
    
    $f_1, f_6, f_8, f_4, f_9, f_5, f_2, f_{11}, f_7, f_{10}, f_3$

\end{homeworkAnswer}
